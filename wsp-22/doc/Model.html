def password_cooldown_counter(cooldown)
    if Time.now.to_i - cooldown > 30
        return true
    else
        return false
    end
end

def password_cooldown_detection(timearray)
    if timearray.length == 3
        if timearray[2] - timearray[0] < 30
            return Time.now.to_i
        else
            return false
        end
    end

    return false
end
